WITH RT AS (
    SELECT DISTINCT 
        TRIM(x.EXTCLM) AS EXTCLM,
        (d.BRANCD || RIGHT(d.BATDAT,4) || SUBSTR(d.BATDAT,2,2) 
         || LPAD(d.BATSEQ,3,'0') || LPAD(d.SEQNUM,2,'0')) 
         || LPAD(d.LINENO,3,'0') AS RT_ReportKey,
        (d.BRANCD || RIGHT(d.BATDAT,4) || SUBSTR(d.BATDAT,2,2) 
         || LPAD(d.BATSEQ,3,'0') || LPAD(d.SEQNUM,2,'0')) AS RT_CLAIM_ID,
        LPAD(d.LINENO,3,'0') AS RT_CLAIM_LINE,
        d.SVCCOD, d.ALWAMT, '' AS ALL_HOLDCDS,

        -- Hold codes 1–6
        (SELECT DISTINCT HOLDCD FROM MHSFLP.EXHMAS 
         WHERE BRANCD=d.BRANCD AND BATDAT=d.BATDAT AND BATSEQ=d.BATSEQ 
           AND CLMSEQ=d.SEQNUM AND LINENO=d.LINENO AND TRNSEQ=1 
         FETCH FIRST 1 ROWS ONLY) AS HOLDCD1,
        (SELECT DISTINCT HOLDCD FROM MHSFLP.EXHMAS 
         WHERE BRANCD=d.BRANCD AND BATDAT=d.BATDAT AND BATSEQ=d.BATSEQ 
           AND CLMSEQ=d.SEQNUM AND LINENO=d.LINENO AND TRNSEQ=2 
         FETCH FIRST 1 ROWS ONLY) AS HOLDCD2,
        (SELECT DISTINCT HOLDCD FROM MHSFLP.EXHMAS 
         WHERE BRANCD=d.BRANCD AND BATDAT=d.BATDAT AND BATSEQ=d.BATSEQ 
           AND CLMSEQ=d.SEQNUM AND LINENO=d.LINENO AND TRNSEQ=3 
         FETCH FIRST 1 ROWS ONLY) AS HOLDCD3,
        (SELECT DISTINCT HOLDCD FROM MHSFLP.EXHMAS 
         WHERE BRANCD=d.BRANCD AND BATDAT=d.BATDAT AND BATSEQ=d.BATSEQ 
           AND CLMSEQ=d.SEQNUM AND LINENO=d.LINENO AND TRNSEQ=4 
         FETCH FIRST 1 ROWS ONLY) AS HOLDCD4,
        (SELECT DISTINCT HOLDCD FROM MHSFLP.EXHMAS 
         WHERE BRANCD=d.BRANCD AND BATDAT=d.BATDAT AND BATSEQ=d.BATSEQ 
           AND CLMSEQ=d.SEQNUM AND LINENO=d.LINENO AND TRNSEQ=5 
         FETCH FIRST 1 ROWS ONLY) AS HOLDCD5,
        (SELECT DISTINCT HOLDCD FROM MHSFLP.EXHMAS 
         WHERE BRANCD=d.BRANCD AND BATDAT=d.BATDAT AND BATSEQ=d.BATSEQ 
           AND CLMSEQ=d.SEQNUM AND LINENO=d.LINENO AND TRNSEQ=6 
         FETCH FIRST 1 ROWS ONLY) AS HOLDCD6,

        -- Other columns unchanged...
        d.FEECOD, d.ALWAMT AS ALLOWED_AMT, d.CLAAMT AS CHARGES,
        d.COMPNO, d.CLATYP, d.POSCOD, d.BENPKG, d.PROVNO, p.PRVTYP,
        TRIM(p.LSTNAM) || ' ' || TRIM(p.FSTNAM) AS CONTRACT_NAME,
        p.FEDNUM AS TIN, d.TOPAMT, d.DEDAMT, d.COPAMT, d.COIAMT, d.OOPAMT,
        d.CPTCOD, d.MODCOD, d.MODCD2, d.MODCD3, d.CPTMD1, d.CPTMD2, d.CPTMD3,
        x.EDIBAT, d.FORMCD, x.STATCD AS ClaimStatus, d.BRANCD, d.BATDAT,
        d.BATSEQ, d.SEQNUM, d.LINENO, d.STATCD AS LineStatus, d.LOBCOD,
        d.MEMBNO, x.CLMFRQ, d.BENCOD, d.SVCDAT, d.ENDDAT, x.DIAGCD001,
        d.PRVCPY, d.SYSDAT, d.EDTDAT, d.PSTDAT, d.UNITCT,
        (CASE WHEN FORMCD='U' THEN 'Facility Claim'
              WHEN FORMCD='H' THEN 'Professional Claim'
              ELSE 'Error' END) AS ClaimType
    FROM MHSFLP.MEDMAS AS d
    JOIN (MHSFLP.MEXMAS AS x 
          JOIN MHSFLP.CLMBCHET AS c ON x.EDIBAT=c.EDIBAT)
      ON d.BRANCD=x.BRANCD AND d.BATDAT=x.BATDAT 
         AND d.BATSEQ=x.BATSEQ AND d.SEQNUM=x.SEQNUM
    LEFT OUTER JOIN MHSFLP.PRVMAS AS p ON d.PROVNO=p.PROVNO
    LEFT OUTER JOIN MHSFLP.APIMAS AS k ON d.PROVNO=k.PROVNO
    LEFT OUTER JOIN MHSFLP.MXCMAS AS m 
      ON d.BRANCD=m.BRANCD AND d.BATDAT=m.BATDAT 
         AND d.BATSEQ=m.BATSEQ AND d.SEQNUM=m.SEQNUM
)
SELECT DISTINCT *,
       CASE 
         WHEN HOLDCD1 = 'VQ' THEN 'HoldCode1'
         WHEN HOLDCD2 = 'VQ' THEN 'HoldCode2'
         WHEN HOLDCD3 = 'VQ' THEN 'HoldCode3'
         WHEN HOLDCD4 = 'VQ' THEN 'HoldCode4'
         WHEN HOLDCD5 = 'VQ' THEN 'HoldCode5'
         WHEN HOLDCD6 = 'VQ' THEN 'HoldCode6'
       END AS MatchedHoldCode
FROM RT
WHERE BATDAT > '2260510'
  AND (
       HOLDCD1 = 'VQ'
    OR HOLDCD2 = 'VQ'
    OR HOLDCD3 = 'VQ'
    OR HOLDCD4 = 'VQ'
    OR HOLDCD5 = 'VQ'
    OR HOLDCD6 = 'VQ'
  )
ORDER BY EXTCLM, BATSEQ, SEQNUM, LINENO;
